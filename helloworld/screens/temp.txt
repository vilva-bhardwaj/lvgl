#include "lvgl.h"

static lv_theme_t * theme_light;
static lv_theme_t * theme_dark;

static void switch_event_cb(lv_event_t * e)
{
    lv_obj_t * sw = lv_event_get_target(e);
    lv_disp_t * disp = lv_disp_get_default();

    bool dark = lv_obj_has_state(sw, LV_STATE_CHECKED);

    if(dark) {
        lv_disp_set_theme(disp, theme_dark);
    } else {
        lv_disp_set_theme(disp, theme_light);
    }

    /* Force refresh */
    lv_obj_clean(lv_scr_act());

    /* Recreate your screen widgets here */
    create_main_screen();
}

void create_main_screen(void)
{
    lv_obj_t * sw = lv_switch_create(lv_scr_act());
    lv_obj_center(sw);

    lv_obj_add_event_cb(sw, switch_event_cb,
                        LV_EVENT_VALUE_CHANGED, NULL);
}

void ui_init(void)
{
    lv_disp_t * disp = lv_disp_get_default();

    theme_light = lv_theme_default_init(
        disp,
        lv_palette_main(LV_PALETTE_BLUE),
        lv_palette_main(LV_PALETTE_RED),
        false,
        LV_FONT_DEFAULT);

    theme_dark = lv_theme_default_init(
        disp,
        lv_palette_main(LV_PALETTE_BLUE),
        lv_palette_main(LV_PALETTE_RED),
        true,
        LV_FONT_DEFAULT);

    lv_disp_set_theme(disp, theme_light);

    create_main_screen();
}
